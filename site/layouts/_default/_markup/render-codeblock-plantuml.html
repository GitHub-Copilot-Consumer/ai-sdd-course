{{/* render-codeblock-plantuml.html Hugo render hook for plantuml fenced code blocks. Sends PlantUML source to the Kroki public API via POST at build time, receives an SVG response and embeds it inline in the page. On API failure: logs a warning and falls back to displaying raw source. Spec: plantuml-rendering - PlantUML code block renders as SVG */}} {{- $diagram := .Inner -}} {{- $krokiURL := "https://kroki.io/plantuml/svg" -}} {{- $opts := dict "method" "POST" "body" $diagram "headers" (dict "Content-Type" "text/plain") -}} {{- with try (resources.GetRemote $krokiURL $opts) -}} {{- with .Err -}} {{- warnf "plantuml render hook: Kroki API error – %s" . -}} 
{{ $diagram }}
{{- else with .Value -}} {{- .Content | safeHTML -}} {{- else -}} {{- warnf "plantuml render hook: Kroki returned no content for diagram at %s" $krokiURL -}} 
{{ $diagram }}
{{- end -}} {{- end -}}
