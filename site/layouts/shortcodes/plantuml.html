{{/* plantuml.html Hugo shortcode for embedding PlantUML diagrams. Usage: wrap PlantUML source with the plantuml shortcode tags. The shortcode POSTs the source to kroki.io and embeds the SVG response. Sends PlantUML source to the Kroki public API via POST at build time, receives an SVG response and embeds it inline in the page. On API failure: logs a warning and falls back to displaying raw source. Spec: plantuml-rendering - PlantUML shortcode support */}} {{- $diagram := .Inner -}} {{- $krokiURL := "https://kroki.io/plantuml/svg" -}} {{- $opts := dict "method" "POST" "body" $diagram "headers" (dict "Content-Type" "text/plain") -}} {{- with try (resources.GetRemote $krokiURL $opts) -}} {{- with .Err -}} {{- warnf "plantuml shortcode: Kroki API error – %s" . -}} 
{{ $diagram }}
{{- else with .Value -}} {{- .Content | safeHTML -}} {{- else -}} {{- warnf "plantuml shortcode: Kroki returned no content for diagram at %s" $krokiURL -}} 
{{ $diagram }}
{{- end -}} {{- end -}}
